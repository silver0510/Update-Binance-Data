CREATE TABLE ETHMARKET(
   ID  			 		INT AUTO_INCREMENT      NOT NULL,
   MARKET 			 	VARCHAR (50)            NOT NULL,
   OPENTIME		 		BIGINT          	NOT NULL,
   OPENPRICE 		 		FLOAT(18,8)   		NOT NULL,
   HIGHPRICE		 		FLOAT(18,8)   		NOT NULL,
   LOWPRICE		 		FLOAT(18,8)   		NOT NULL,
   CLOSEPRICE		 		FLOAT(18,8)   		NOT NULL,
   VOLUME		 		FLOAT(18,8)          	NOT NULL,
   CLOSETIME		 		BIGINT            	NOT NULL,
   QUOTE_ASSET_VOLUME	 		FLOAT(18,8)   		NOT NULL,
   NUMBER_OF_TRADES 			INT   			NOT NULL,
   TAKER_BUY_BASE_ASSET_VOLUME		FLOAT(18,8)   		NOT NULL,
   TAKER_BUY_QUOTE_ASSET_VOLUME 	FLOAT(18,8)   		NOT NULL,
   IGNORE_DATA 				FLOAT(18,8)          	NOT NULL,
   CHANGE24H   				FLOAT (10, 3),       
   PRIMARY KEY (ID)
);
